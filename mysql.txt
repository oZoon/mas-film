ручное создание пользователя и БД (от имени root)

CREATE DATABASE `masha`;
FLUSH PRIVILEGES;
CREATE USER 'masha'@'localhost' IDENTIFIED BY '12345';
GRANT USAGE ON *.* TO 'masha'@'localhost';
GRANT EXECUTE, SELECT, SHOW VIEW, ALTER, ALTER ROUTINE, CREATE, CREATE ROUTINE, CREATE TEMPORARY TABLES, CREATE VIEW, DELETE, DROP, EVENT, INDEX, INSERT, REFERENCES, TRIGGER, UPDATE, LOCK TABLES  ON `masha`.* TO 'masha'@'localhost' WITH GRANT OPTION;
FLUSH PRIVILEGES;





на автомате

USE `masha`;
CREATE TABLE `films` (
	`filmID` INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,
	`filmName` VARCHAR(50) NULL DEFAULT NULL,
	`genreID` TINYINT(4) NULL DEFAULT NULL,
	PRIMARY KEY (`filmID`)
)
ENGINE=MyISAM;

CREATE TABLE `actors` (
	`actorID` INT(10) UNSIGNED NOT NULL AUTO_INCREMENT,
	`actorFullName` VARCHAR(50) NULL DEFAULT NULL,
	PRIMARY KEY (`actorID`)
)
ENGINE=MyISAM;

CREATE TABLE `genres` (
	`genreID` TINYINT(3) UNSIGNED NOT NULL AUTO_INCREMENT,
	`genreName` VARCHAR(50) NULL DEFAULT NULL,
	PRIMARY KEY (`genreID`)
)
ENGINE=MyISAM;

CREATE TABLE `links` (
	`filmID` INT(11) NULL DEFAULT NULL,
	`actorID` INT(11) NULL DEFAULT NULL,
	`genreID` TINYINT(4) NULL DEFAULT NULL,
	INDEX `filmID` (`filmID`),
	INDEX `actorID` (`actorID`),
	INDEX `genreID` (`genreID`)
)
ENGINE=MyISAM;

ALTER TABLE `links` ADD CONSTRAINT `FK_links_films` FOREIGN KEY (`filmID`) REFERENCES `films` (`filmID`);

ALTER TABLE `links` ADD CONSTRAINT `FK_links_actors` FOREIGN KEY (`actorID`) REFERENCES `actors` (`actorID`);

ALTER TABLE `links` ADD CONSTRAINT `FK_links_genres` FOREIGN KEY (`genreID`) REFERENCES `genres` (`genreID`);
